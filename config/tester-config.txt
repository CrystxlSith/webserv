server {
    listen 8080;
    server_name localhost;
    root /www/simplesite;
    client_max_body_size 10M;
    error_page 404 /errors/404.html;
    error_page 500 /errors/500.html;

    # / must answer to GET request ONLY
    location / {
        methods GET;
        index index.html;
        client_max_body_size 10M;
        autoindex on;
    }

    # /put_test/* must answer to PUT request and save files to a directory of your choice
    location /put_test {
        methods PUT;
        upload_dir /var/uploads/put_test;
        client_max_body_size 10M;
    }

    # any file with .bla extension must answer to POST request by calling the cgi_test executable
    location ~ \.bla$ {
        methods POST;
        cgi_ext .bla;
        cgi_path /usr/local/bin/cgi_test;
        root /www/simplesite;
    }

    # /post_body must answer anything to POST request with a maxBody of 100
    location /post_body {
        methods POST;
        client_max_body_size 100;
    }

    # /directory/ must answer to GET request with root YoupiBanane and search for youpi.bad_extension files
    location /directory {
        methods GET;
        root /YoupiBanane;
        index youpi.bad_extension;
        autoindex on;
    }
}
